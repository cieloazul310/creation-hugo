```JavaScript
  const d3 = require("d3");

  d3.json("./data/data2016.json", (err, json)=>{
    if(err) throw err;
    const xScale = d3.scaleLinear()
                      .domain([0, d3.max(json.map(d => d.value))]);
    const yScale = d3.scaleOrdinal()
                      .domain(json.map(d => d.name));
  });
```
